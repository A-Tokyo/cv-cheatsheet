\section{Mathematics}
\subsubsection{Cross Product}
\[
m \times n =
\begin{bmatrix}
  m_1\\
  m_2\\
  m_3
\end{bmatrix} \times
\begin{bmatrix}
  n_1\\
  n_2\\
  n_3
\end{bmatrix} =
\begin{bmatrix}
  m_2 m_3 - m_3 m_2\\
  m_3 m_1 - m_1 m_3\\
  m_1 m_2 - m_2 m_1
\end{bmatrix}
\]
